Logging
SAS Viya Standard Log
# Default, plain message without color-coded
%put standard message;
		
# NOTE, UPPER case in blue color
%put NOTE: note message;
		
# WARNING, UPPER case in green color
%put WARNING: warning message;
		
# ERROR, UPPER case in red color
%put ERROR: error message;
Log to File
# redirect log to a file
proc printto log="path/log.txt";
run;
		
# write log
%put NOTE: message;
		
# redirect log to standard output
proc printto;
run;
