For
l = [1, 2, 3, 4]

for i in l:
    print(i, id(i)) # print element and id

for i in l:
    # i is immutable
    # changing it creates a new object, the object in l does not change
    i = i*10
    print(i, id(i))

print(l) # [1, 2, 3, 4]

for index, ele in enumerate(l):
    # l is mutable
    # changing an item creates a new object, and replaces the old object
    l[index] = l[index]*10
    # l[index] = ele*10
    print(index, l[index], id(l[index]), ele, id(ele))

print(l) # [10, 20, 30, 40]
Remove items in for loop 

# wrong
l = [1, 2, 3, 4, 5]
for item in l:
    l.remove(item)

print(l) # [2, 4]
		

# correct
l = [1, 2, 3, 4, 5]
# l[:] is a shadow copy of l
for item in l[:]:
    l.remove(item)

print(l) # []
