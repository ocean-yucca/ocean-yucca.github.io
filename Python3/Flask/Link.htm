Default Route
  • Maps a route path to a function
    • import os
from flask import Flask

app = Flask(__name__)

# default route
@app.route('/')
def p1():
    return """
    <html>
    <body>
        <h1>Hello World!</h1>
    </body>
    </html>
    """

if __name__ == '__main__':
    app.run(port = 8080)

# access
# http://localhost:8080/
    Multiple Routes
  • Maps multiple route pathes to a function
    • import os
from flask import Flask

app = Flask(__name__)

@app.route('/')
@app.route('/home')
def p1():
    return """
    <html>
    <body>
        <h1>Hello World!</h1>
    </body>
    </html>
    """

if __name__ == '__main__':
    app.run(port = 8080)

# access
# http://localhost:8080/
# http://localhost:8080/home
    Links between Pages
    import os
from flask import Flask

app = Flask(__name__)

@app.route('/')
def p1():
    return """
    <html>
    <body>
        <h1>Page 1</h1>
        Link to: <a href = "/page_2">Page 2</a>
    </body>
    </html>
    """

@app.route('/page_2')
def p2():
    return """
    <html>
    <body>
        <h1>Page 2</h1>
        Link to: <a href = "/">Page 1</a>
    </body>
    </html>
    """

if __name__ == '__main__':
    app.run(port = 8080)

    Function Name to Route
  • url_for, convert function name to route
    • from flask import Flask, url_for, redirect

app = Flask(__name__)

@app.route('/hello')
def my_hello():
    return "Hello!"

@app.route('/')
def index():
    return redirect(url_for('my_hello')) # /hello

if __name__ == '__main__':
    app.run(port = 8080)
