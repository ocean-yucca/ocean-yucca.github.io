Layout
  • templates, default location where Flask looks for HTML (Jinja2) template files used to render dynamic web pages
  • static, default directory used to serve static files such as css files, javascript files, images, pdf files, etc.

    • # app.py
import os
from flask import Flask, render_template, url_for

app = Flask(__name__)

@app.route("/")
def hello_name():
    return render_template('index.html')

if __name__ == '__main__':
    app.run(port = 8080)
        
    # index.html
<!doctype html>
<html>
    <head>
        <script type = "text/javascript" src = "{{ url_for('static', filename='js/hello.js') }}"></script>
        <link rel="stylesheet" href="{{ url_for('static', filename='css/style.css') }}">
    </head>
    <body>
        <img src = "{{ url_for('static', filename='img/profile.jpg') }}" />
        <input type = "button" onclick="sayHello()" value = "Say Hello" />
    </body>
</html>
        
    # style.css
body {
    background-color: lightblue;
}
        
    # hello.js
function sayHello() {
    alert("Hello World!")
}
