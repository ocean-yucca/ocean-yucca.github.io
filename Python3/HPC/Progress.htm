import time

weights = [10, 20, 30, 10, 20, 40]

with tqdm(total=sum(weights)) as pbar:
    for weight in weights:
        time.sleep(0.1)
        pbar.update(weight)

from rich.progress import Progress
import time

with Progress(transient=True) as progress:
    task_a = progress.add_task("Downloading A", total=100)
    task_b = progress.add_task("Downloading B", total=100)

    while not progress.finished:
        progress.update(task_a, advance=1)
        progress.update(task_b, advance=2)
        time.sleep(0.05)
