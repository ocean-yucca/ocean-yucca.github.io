Management
Create Compose File
services:
  neo4j:
    image: neo4j:5.20
    container_name: neo4j
    ports:
      - "7474:7474"   # HTTP / Browser
      - "7687:7687"   # Bolt (driver connections)
    environment:
      NEO4J_AUTH: neo4j/test1234
      NEO4J_PLUGINS: '["apoc"]'
      NEO4J_dbms_security_procedures_unrestricted: apoc.*
    volumes:
      - neo4j_data:/data
    restart: unless-stopped

volumes:
  neo4j_data:
Start Neo4j
docker-compose up # start Neo4j and Neo4j Browser

#docker-compose down # stop Neo4j and Neo4j Browser
Access Neo4j Browser
http://localhost:7474 # neo4j, test1234
